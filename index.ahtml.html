    <script>

        const apiKey = "";

        const traits = ["Stoicism", "Curiosity", "Nostalgia", "Ambition", "Empathy", "Skepticism", "Wonder", "Resilience", "Solitude", "Legacy", "Justice", "Artistry"];

        let selectedTraits = [];

        let base64Image = null;

        let agedBase64 = null;

        let futurePersona = "";

        let currentSessionId = null;

 

        // Ensure Lucide icons are initialized when document is ready

        document.addEventListener('DOMContentLoaded', () => {

            if (typeof lucide !== 'undefined') {

                lucide.createIcons();

            }

        });

 

        window.onload = async () => {

            initTraits();

            checkSharedLink();

        };

 

        function initTraits() {

            const container = document.getElementById('trait-container');

            if (!container) return;

            traits.forEach(trait => {

                const btn = document.createElement('button');

                btn.className = 'trait-chip px-4 py-2 border border-gray-600 rounded-full text-sm text-gray-400 hover:border-white hover:text-white';

                btn.innerText = trait;

                btn.onclick = () => toggleTrait(trait, btn);

                container.appendChild(btn);

            });

        }

 

        async function checkSharedLink() {

            const urlParams = new URLSearchParams(window.location.search);

            const sessionId = urlParams.get('synthesis');

            if (sessionId) {

                currentSessionId = sessionId;

                loadSharedSession(sessionId);

            }

        }

 

        async function loadSharedSession(id) {

            goToStep('step-call');

            document.getElementById('loading-status').innerText = "Retrieving reflection from the cloud...";

           

            try {

                // Wait for firebaseStore to be attached to window by the module script

                let attempts = 0;

                while (!window.firebaseStore && attempts < 50) {

                    await new Promise(r => setTimeout(r, 100));

                    attempts++;

                }

 

                if (!window.firebaseStore) throw new Error("Firebase not initialized");

 

                const { db, appId, doc, getDoc, auth } = window.firebaseStore;

                // Ensure auth

                if (!auth.currentUser) {

                    const { signInAnonymously } = await import("https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js");

                    await signInAnonymously(auth);

                }

 

                const docRef = doc(db, 'artifacts', appId, 'public', 'data', 'sessions', id);

                const docSnap = await getDoc(docRef);

 

                if (docSnap.exists()) {

                    const data = docSnap.data();

                    base64Image = data.originalImage;

                    agedBase64 = data.agedImage;

                    selectedTraits = data.traits;

                   

                    document.getElementById('local-view').src = `data:image/png;base64,${base64Image}`;

                    const img = document.getElementById('future-self-img');

                    img.src = `data:image/png;base64,${agedBase64}`;

                    img.classList.remove('hidden');

                    document.getElementById('loading-future').classList.add('hidden');

                   

                    setupPersona();

                    showSubtitle("The connection is restored. We were just speaking...");

                } else {

                    alert("This reflection has faded into the labyrinth.");

                    location.href = window.location.pathname;

                }

            } catch (err) {

                console.error(err);

                alert("Error connecting to the cloud.");

            }

        }

 

        function toggleTrait(trait, el) {

            if (selectedTraits.includes(trait)) {

                selectedTraits = selectedTraits.filter(t => t !== trait);

                el.classList.remove('active');

            } else {

                selectedTraits.push(trait);

                el.classList.add('active');

            }

        }

 

        function goToStep(stepId) {

            document.querySelectorAll('#app > div').forEach(div => div.classList.add('hidden'));

            document.getElementById(stepId).classList.remove('hidden');

        }

 

        function handlePhotoUpload(event) {

            const file = event.target.files[0];

            if (file) {

                const reader = new FileReader();

                reader.onload = (e) => {

                    base64Image = e.target.result.split(',')[1];

                    const preview = document.getElementById('photo-preview');

                    preview.src = e.target.result;

                    preview.classList.remove('hidden');

                    document.getElementById('upload-placeholder').classList.add('hidden');

                    document.getElementById('local-view').src = e.target.result;

                };

                reader.readAsDataURL(file);

            }

        }

 

        function setupPersona() {

            futurePersona = `You are the future version of the user, 25 years older.

            Values: ${selectedTraits.join(', ')}. Tone: Borgesian, poetic, wise.`;

        }

 

        async function startCall() {

            if (!base64Image || selectedTraits.length === 0) return alert("Photo and traits required.");

            goToStep('step-call');

            setupPersona();

            await generateFutureReflection();

            await generateOpeningStatement();

        }

 

        async function generateFutureReflection() {

            try {

                const prompt = "A hyper-realistic cinematic portrait of this same person but 25 years older, with realistic skin aging, graying hair, wisdom in the eyes.";

                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-image-preview:generateContent?key=${apiKey}`, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify({

                        contents: [{ parts: [{ text: prompt }, { inlineData: { mimeType: "image/png", data: base64Image } }] }],

                        generationConfig: { responseModalities: ['TEXT', 'IMAGE'] }

                    })

                });

                const result = await response.json();

                agedBase64 = result.candidates?.[0]?.content?.parts?.find(p => p.inlineData)?.inlineData?.data;

                if (agedBase64) {

                    const img = document.getElementById('future-self-img');

                    img.src = `data:image/png;base64,${agedBase64}`;

                    img.classList.remove('hidden');

                    document.getElementById('loading-future').classList.add('hidden');

                }

            } catch (err) { console.error(err); }

        }

 

        async function generateOpeningStatement() {

            try {

                const res = await callGeminiLLM(`Greet your younger self. Be brief and poetic.`, true);

                const text = res.candidates?.[0]?.content?.parts?.[0]?.text || "I am here.";

                showSubtitle(text);

            } catch (err) {

                console.error(err);

            }

        }

 

        async function callGeminiLLM(query, includeImage = false) {

            const parts = [{ text: query }];

            if (includeImage) parts.push({ inlineData: { mimeType: "image/png", data: base64Image } });

            const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {

                method: 'POST',

                headers: { 'Content-Type': 'application/json' },

                body: JSON.stringify({

                    contents: [{ parts: parts }],

                    systemInstruction: { parts: [{ text: futurePersona }] }

                })

            });

            return response.json();

        }

 

        async function shareSynthesis() {

            const btn = document.getElementById('share-btn');

            btn.disabled = true;

            showSubtitle("Preserving this reflection in the cloud...");

 

            try {

                const { db, appId, doc, setDoc, auth } = window.firebaseStore;

                if (!auth.currentUser) {

                    const { signInAnonymously } = await import("https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js");

                    await signInAnonymously(auth);

                }

 

                const sessionId = crypto.randomUUID();

                await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'sessions', sessionId), {

                    originalImage: base64Image,

                    agedImage: agedBase64,

                    traits: selectedTraits,

                    createdAt: new Date().toISOString()

                });

 

                const shareUrl = `${window.location.origin}${window.location.pathname}?synthesis=${sessionId}`;

               

                try {

                    await navigator.clipboard.writeText(shareUrl);

                } catch (clipErr) {

                    // Fallback

                    const textArea = document.createElement("textarea");

                    textArea.value = shareUrl;

                    document.body.appendChild(textArea);

                    textArea.select();

                    document.execCommand('copy');

                    document.body.removeChild(textArea);

                }

               

                showSubtitle("Reflection saved. The link is copied to your clipboard. Share it to return to this moment.");

                btn.disabled = false;

            } catch (err) {

                console.error(err);

                showSubtitle("The cloud is heavy with rain. Could not save reflection.");

                btn.disabled = false;

            }

        }

 

        function askQuestion() {

            const q = prompt("What would you like to ask your older self?");

            if (!q) return;

            showSubtitle("Considering...");

            callGeminiLLM(q).then(res => {

                const text = res.candidates?.[0]?.content?.parts?.[0]?.text || "The stars are silent on this.";

                showSubtitle(text);

            }).catch(err => {

                showSubtitle("A ripple in the connection.");

            });

        }

 

        function showSubtitle(text) {

            const box = document.getElementById('subtitle-box');

            if (box) {

                box.innerText = text;

                box.style.opacity = '1';

            }

        }

 

        function endCall() {

            if (confirm("Reset the synthesis?")) location.href = window.location.pathname;

        }
